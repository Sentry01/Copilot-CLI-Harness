<project_specification>
  <project_name>Copilot CLI Guide - Interactive Learning Platform</project_name>

  <overview>
    Build an interactive educational web application that teaches users how to effectively use GitHub Copilot CLI.
    The application provides step-by-step tutorials, interactive command sandboxes, real examples, and a 
    comprehensive reference guide. Users can learn concepts progressively, practice commands in simulated 
    environments, and track their learning progress. The UI should be clean, modern, and developer-focused
    using Tailwind CSS with excellent code presentation and responsive design.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React with Vite</framework>
      <styling>Tailwind CSS (via CDN)</styling>
      <state_management>React hooks and context</state_management>
      <routing>React Router for navigation</routing>
      <markdown>React Markdown for content rendering</markdown>
      <code_highlighting>Prism.js or Highlight.js for syntax highlighting</code_highlighting>
      <animations>Framer Motion for smooth transitions</animations>
      <port>Launch on port 5173</port>
    </frontend>
    <backend>
      <runtime>Node.js with Express</runtime>
      <database>SQLite with better-sqlite3</database>
      <content>Markdown files for tutorial content</content>
    </backend>
    <communication>
      <api>RESTful endpoints</api>
      <real_time>WebSocket for live terminal simulation</real_time>
    </communication>
  </technology_stack>

      - GET /api/examples/:id - Get example details

      - POST /api/progress/lesson/:id - Mark lesson complete

      - GET /api/progress/stats - Get learning statistics
    </progress>

    <bookmarks>
      - GET /api/bookmarks - List user's bookmarks
      - POST /api/bookmarks - Create bookmark
      - DELETE /api/bookmarks/:id - Remove bookmark

    </bookmarks>

    <achievements>
      - GET /api/achievements - List all achievements

    </search>

    <user>

      - PUT /api/user/preferences - Update preferences
